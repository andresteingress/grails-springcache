016    package grails.plugin.springcache.key;
017    
018    import grails.plugin.springcache.CacheKey;
019    
020    /**
021     * Builder used to compute the hash and checksum used for an immutable cache key.
022     * <p/>
023     * Based on org.springmodules.cache.key.HashCodeCalculator from Spring Modules Cache (see
024     * https://springmodules.dev.java.net/) by Omar Irbouh and Alex Ruiz.
025     */
026    public class CacheKeyBuilder {
027    
028            private static final int INITIAL_HASH = 17;
029            private static final int MULTIPLIER = 31;
030    
031            private int count = 0;
032            private long checksum = 0L;
033            private int hash = INITIAL_HASH;
034    
035            public CacheKeyBuilder append(int i) {
036                    hash = hash * MULTIPLIER + i;
037                    count++;
038                    checksum += (count * i);
039                    return this;
040            }
041    
042            public CacheKeyBuilder append(Object o) {
043                    return append(o.hashCode());
044            }
045    
046            public CacheKeyBuilder append(Object[] oarr) {
047                    for (Object o : oarr) {
048                            append(o.hashCode());
049                    }
050                    return this;
051            }
052    
053            public void leftShift(Object o) {
054                    append(o);
055            }
056    
057            public void leftShift(Object[] oarr) {
058                    append(oarr);
059            }
060    
061            public CacheKey toCacheKey() {
062                    return new CacheKey(hash, checksum);
063            }
064    
065    }
