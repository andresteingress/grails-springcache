016    package grails.plugin.springcache.key;
017    
018    import grails.plugin.springcache.CacheKey;
019    import java.util.Arrays;
020    
021    /**
022     * Builder used to compute the hash and checksum used for an immutable cache key.
023     * <p/>
024     * Based on org.springmodules.cache.key.HashCodeCalculator from Spring Modules Cache (see
025     * https://springmodules.dev.java.net/) by Omar Irbouh and Alex Ruiz.
026     */
027    public class CacheKeyBuilder {
028    
029            private static final int INITIAL_HASH = 17;
030            private static final int MULTIPLIER = 31;
031    
032            private int count = 0;
033            private long checksum = 0L;
034            private int hash = INITIAL_HASH;
035    
036            public CacheKeyBuilder append(int i) {
037                    hash = hash * MULTIPLIER + i;
038                    count++;
039                    checksum += (count * i);
040                    return this;
041            }
042    
043            public CacheKeyBuilder append(Object o) {
044                    if (o == null) {
045                            appendNull();
046                    } else if (o.getClass().isArray()) {
047                            if (o instanceof Object[]) {
048                                    append(Arrays.deepHashCode((Object[])o)); // This will fail if o contains itself
049                            } else if (o instanceof boolean[]) {
050                                    append(Arrays.hashCode((boolean[])o));
051                            } else if (o instanceof byte[]) {
052                                    append(Arrays.hashCode((byte[])o));
053                            } else if (o instanceof char[]) {
054                                    append(Arrays.hashCode((char[])o));
055                            } else if (o instanceof double[]) {
056                                    append(Arrays.hashCode((double[])o));
057                            } else if (o instanceof float[]) {
058                                    append(Arrays.hashCode((float[])o));
059                            } else if (o instanceof int[]) {
060                                    append(Arrays.hashCode((int[])o));
061                            } else if (o instanceof long[]) {
062                                    append(Arrays.hashCode((long[])o));
063                            } else if (o instanceof short[]) {
064                                    append(Arrays.hashCode((short[])o));
065                            } else {
066                                    throw new UnHandleableCacheKeyComponentException(o.getClass());
067                            }
068                    } else {
069                            append(o.hashCode());
070                    }
071    
072                    return this;
073            }
074    
075            public CacheKeyBuilder append(Object[] oarr) {
076                    for (Object o : oarr) {
077                            append(o);
078                    }
079                    return this;
080            }
081    
082            protected CacheKeyBuilder appendNull() {
083                    // Unsure what the best strategy is hereâ€¦
084                    // Append 0 or just do nothing?
085                    
086                    return this;
087            }
088    
089            public void leftShift(Object o) {
090                    append(o);
091            }
092    
093            public void leftShift(Object[] oarr) {
094                    append(oarr);
095            }
096    
097            public CacheKey toCacheKey() {
098                    return new CacheKey(hash, checksum);
099            }
100    
101            protected static class UnHandleableCacheKeyComponentException extends IllegalStateException {
102                    public UnHandleableCacheKeyComponentException(Class type) {
103                            super("Unable to handle object of type '" + type + "'");
104                    }
105            }
106    }
